       ```python
       class TraefikAnyRouterFailingBinarySensor(TraefikEntity, BinarySensorEntity):
           """Aggregates router health: ON when ANY router status != 'enabled' (CONTEXT.md D-14).

           Single instance per config entry — never deleted (CONTEXT.md D-19).
           Lives on the Diagnostics device. entity_registry_enabled_default=False
           (PITFALLS M-12) so it does not pollute the States panel by default.
           """

           _attr_entity_registry_enabled_default = False

           def __init__(
               self,
               entry: TraefikConfigEntry,
               coordinator: TraefikCoordinator,
           ) -> None:
               super().__init__(entry, category="diagnostics", description_key="any_router_failing")
               self._attr_unique_id = f"{entry.entry_id}_diagnostics_any_router_failing"
               self.entity_id = "binary_sensor.traefik_any_router_failing"
               self._attr_name = "Any router failing"

           @property
           def is_on(self) -> bool | None:
               routers = self.coordinator.data.get("http_routers") if isinstance(self.coordinator.data, dict) else None
               if routers is None:
                   return None
               failing = [r for r in routers if r.get("status") != "enabled"]

           @property
           def extra_state_attributes(self) -> dict[str, Any]:
               routers = self.coordinator.data.get("http_routers") if isinstance(self.coordinator.data, dict) else None
               failing = [r for r in (routers or []) if r.get("status") != "enabled"]
               return {
                   "failing_router_count": len(failing),
                   "failing_router_names": [r.get("name") for r in failing],
               }

           @property
           def available(self) -> bool:
               return self.coordinator.last_update_success
       ```
