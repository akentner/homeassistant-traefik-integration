       ```python
       async def _async_handle_reload_routers(hass: HomeAssistant, call: ServiceCall) -> dict[str, Any]:
           """Reload Traefik dynamic config and verify completion via router-set polling.

           CONTEXT.md D-12: Traefik returns 200 from /api/http/routers/refresh BEFORE
           providers finish reloading, so we capture the pre-POST router-name set,
           POST the refresh, then poll coordinator.data['http_routers'] with
           exponential backoff (200ms -> 5s, max 10 attempts, <= 5s budget) and
           exit early when the name set changes from the snapshot.

           Returns: {verified, elapsed_ms, attempts, name_diff}
           """
           started = _LOGGER.debug and None  # placeholder; remove
           import time
           t0 = time.monotonic()

           # Find the first loaded Traefik coordinator. The integration is `service` type
           # (one Traefik per config entry), so there should be exactly one loaded entry.
           coordinators = [
               entry.runtime_data
               for entry in hass.config_entries.async_entries(DOMAIN)
               if entry.state is ConfigEntryState.LOADED
           ]
           if not coordinators:
               raise HomeAssistantError("No loaded Traefik config entry to reload against")
           coordinator: TraefikCoordinator = coordinators[0]

           before = {r["name"] for r in (coordinator.data or {}).get("http_routers", []) if isinstance(r, dict)}

           # Trigger the refresh. Raises TraefikApiError -> HA surfaces as service-call failure.
           await coordinator.client.reload_routers()

           # Poll the coordinator for an observable change.
           attempts = 0
           verified = False
           backoff_ms = 200
           max_attempts = 10
           max_budget_ms = 5000
           while attempts < max_attempts and (time.monotonic() - t0) * 1000 < max_budget_ms:
               attempts += 1
               await coordinator.async_request_refresh()
               # Wait for the refresh cycle to finish.
               try:
                   await asyncio.wait_for(
                       coordinator.async_wait_for_ready(),
                       timeout=2.0,
                   )
               except (asyncio.TimeoutError, Exception):
                   # Coordinator didn't reach ready in time; continue polling.
                   pass
               current = {r["name"] for r in (coordinator.data or {}).get("http_routers", []) if isinstance(r, dict)}
               if current != before:
                   verified = True
                   break
               await asyncio.sleep(backoff_ms / 1000)
               backoff_ms = min(backoff_ms * 2, 5000)

           elapsed_ms = int((time.monotonic() - t0) * 1000)
           name_diff = {
               "added": sorted((current - before)) if verified else [],
               "removed": sorted((before - current)) if verified else [],
           }
           _LOGGER.debug(
               "traefik.reload_routers: verified=%s elapsed_ms=%d attempts=%d name_diff=%s",
               verified, elapsed_ms, attempts, name_diff,
           )
           return {
               "verified": verified,
               "elapsed_ms": elapsed_ms,
               "attempts": attempts,
               "name_diff": name_diff,
           }
       ```
